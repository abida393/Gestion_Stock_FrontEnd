================================================================
  STOCK MANAGER — API ENDPOINT LIST (v1)
================================================================

Base URL: /api/v1
Auth: Protected by auth:sanctum (Bearer Token) unless specified

----------------------------------------------------------------
 AUTHENTICATION (/auth)
----------------------------------------------------------------
POST   /register                        Create new user account
POST   /login                           Authenticate, returns Bearer token
POST   /logout                          Revoke current token (auth)
GET    /me                              Get authenticated user profile (auth)
PUT    /me                              Update own profile / password (auth)

----------------------------------------------------------------
 USER MANAGEMENT (/users)
----------------------------------------------------------------
GET    /users                           List all users (paginated) (auth)
POST   /users                           Create user (admin) (auth)
GET    /users/{id}                      Get user details (auth)
PUT    /users/{id}                      Update user (auth)
DELETE /users/{id}                      Delete user (auth)
POST   /users/{id}/roles                Assign role to user (auth)
DELETE /users/{id}/roles/{roleId}       Remove role from user (auth)

----------------------------------------------------------------
 RBAC (/roles, /permissions)
----------------------------------------------------------------
GET    /roles                           List all roles (auth)
POST   /roles                           Create role (auth)
GET    /roles/{id}                      Get role details (auth)
PUT    /roles/{id}                      Update role name (auth)
DELETE /roles/{id}                      Delete role (auth)
POST   /roles/{id}/permissions          Sync permissions for a role (auth)
GET    /permissions                     List available permissions (auth)

----------------------------------------------------------------
 INVENTORY (/categories, /products)
----------------------------------------------------------------
GET    /categories                      List all categories (auth)
POST   /categories                      Create category (auth)
GET    /categories/{id}                 Get category with products (auth)
PUT    /categories/{id}                 Update category (auth)
DELETE /categories/{id}                 Delete category (auth)

GET    /products                        List all products (filters: search, cat) (auth)
POST   /products                        Create product (with image support) (auth)
GET    /products/{id}                   Get product details (auth)
PUT    /products/{id}                   Update product (auth)
DELETE /products/{id}                   Soft delete product (auth)
POST   /products/{id}/image             Upload product image (auth)
GET    /products/low-stock              List products below minimum threshold (auth)

----------------------------------------------------------------
 SUPPLY CHAIN (/fournisseurs, /commandes)
----------------------------------------------------------------
GET    /fournisseurs                   List all suppliers (auth)
POST   /fournisseurs                   Create supplier (auth)
GET    /fournisseurs/{id}              Get supplier details (auth)
PUT    /fournisseurs/{id}              Update supplier (auth)
DELETE /fournisseurs/{id}              Soft delete supplier (auth)

GET    /commandes                      List all orders (filters: status, vendor) (auth)
POST   /commandes                      Create order with lines (atomic) (auth)
GET    /commandes/{id}                 Get order with line items (auth)
PUT    /commandes/{id}                 Update order (if en_attente) (auth)
DELETE /commandes/{id}                 Cancel/Delete order (auth)
PATCH  /commandes/{id}/statut          Change state (en_attente/livree/annulee) (auth)

----------------------------------------------------------------
 MOVEMENTS & ALERTS (/mouvements, /alertes)
----------------------------------------------------------------
GET    /mouvements                      List all stock movements (auth)
POST   /mouvements                      Record movement (auto-updates stock) (auth)
GET    /mouvements/{id}                 Get movement details (auth)
GET    /mouvements/entrees              List only entries (auth)
GET    /mouvements/sorties              List only exits (auth)

GET    /alertes                        List all alerts (auth)
GET    /alertes/actives                List current active alerts (auth)
PATCH  /alertes/{id}/resolve           Mark alert as resolved (auth)
DELETE /alertes/{id}                   Delete alert record (auth)

----------------------------------------------------------------
 NOTIFICATIONS (/notifications)
----------------------------------------------------------------
GET    /notifications                 Get current user's notifications (auth)
GET    /notifications/unread-count    Get count of unread notifications (auth)
PATCH  /notifications/read-all        Mark all as read (auth)
PATCH  /notifications/{id}/read       Mark specific notification as read (auth)
DELETE /notifications/{id}            Delete notification (auth)

----------------------------------------------------------------
 REPORTING & DASHBOARD (/dashboard, /rapports, /previsions)
----------------------------------------------------------------
GET    /dashboard                      Get global KPIs (Value, Low Stock, etc) (auth)
GET    /rapports                       List generated reports (auth)
POST   /rapports                       Save report metadata (auth)
GET    /rapports/{id}/download         Download report file (auth)
DELETE /rapports/{id}                 Delete report record (auth)

GET    /previsions                     List demand forecasts (auth)
POST   /previsions                     Create new forecast (auth)
DELETE /previsions/{id}                Delete forecast record (auth)
GET    /historique-ventes              Get sales history for analytics (auth)
POST   /historique-ventes              Record manual sales entry (auth)
